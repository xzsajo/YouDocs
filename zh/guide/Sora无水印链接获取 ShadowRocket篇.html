获取飞书多维表格Webhook链接 ​

创建GitHub Gist ​

  • 访问 GitHub Gist 并创建一个新的 Gist。
  • 命名为 soralink2bitable.js
  • 填入以下内容，注意替换[第三步获取的飞书多维表格Webhook链接]为你实际获取到的链接
javascript
// ===================================================
// Sora Link to Bitable - Surge 脚本
// 版本：v1.1.0
// 更新时间：2025-10-01
// 作者：wycbug
// ===================================================
//
// 更新日志：
// v1.1.0 (2025-10-01) - 添加视频链接筛选，过滤图片格式（.webp等）
// v1.0.0 (2025-10-01) - 初始版本，支持提取Sora视频链接并发送到webhook
//
// Surge文档：https://manual.nssurge.com/scripting/common.html
//
// Surge HTTP Request 脚本 - 提取 Sora 视频链接并发送到飞书 webhook
//
// 配置方法：在 Surge 配置文件的 [Script] 部分添加：
// sora-webhook = type=http-request,pattern=^https?://videos\.openai\.com/,requires-body=0,script-path=js/soralink2bitable.js,debug=1
//
// 说明：此脚本会拦截所有 videos.openai.com 的请求，提取视频链接并发送到飞书webhook
// 重要：需要在 MITM 的 hostname 中包含 *.openai.com，并确保不在 hostname-disabled 中

const SCRIPT_VERSION = "v1.1.0";
console.log("=== Sora 视频链接提取脚本启动 ===");
console.log("脚本版本:", SCRIPT_VERSION);

// Webhook 配置
const WEBHOOK_URL =
  "[第三步获取的飞书多维表格Webhook链接]";

// 获取请求URL
const requestUrl = $request.url;
console.log("捕获到请求URL:", requestUrl);

// 视频格式白名单
const VIDEO_EXTENSIONS = [
  ".mp4",
  ".mov",
  ".avi",
  ".mkv",
  ".webm",
  ".flv",
  ".m4v",
];
// 图片格式黑名单
const IMAGE_EXTENSIONS = [
  ".webp",
  ".jpg",
  ".jpeg",
  ".png",
  ".gif",
  ".bmp",
  ".svg",
  ".ico",
];

// 检查是否是视频链接的函数
function isVideoUrl(url) {
  const urlLower = url.toLowerCase();

  // 排除图片格式
  for (const ext of IMAGE_EXTENSIONS) {
    if (urlLower.includes(ext)) {
      console.log(`→ 检测到图片格式 ${ext}，跳过`);
      return false;
    }
  }

  // 检查是否包含视频格式
  for (const ext of VIDEO_EXTENSIONS) {
    if (urlLower.includes(ext)) {
      console.log(`✓ 检测到视频格式 ${ext}`);
      return true;
    }
  }

  // 如果URL路径包含 /videos/ 也认为是视频
  if (urlLower.includes("/videos/")) {
    console.log("✓ URL路径包含 /videos/");
    return true;
  }

  console.log("→ 未识别为视频链接");
  return false;
}

// 检查是否是 Sora 视频链接
if (requestUrl && requestUrl.includes("videos.openai.com")) {
  console.log("检测到 videos.openai.com 域名");

  // 进一步检查是否是视频链接
  if (!isVideoUrl(requestUrl)) {
    console.log("→ 非视频链接，跳过处理");
  } else {
    console.log("✓ 确认为 Sora 视频链接");

    // 提取视频名称（从URL中提取task_id作为名称）
    let videoName = "Sora视频";
    const taskIdMatch = requestUrl.match(/task_([a-z0-9]+)/);
    if (taskIdMatch) {
      videoName = taskIdMatch[0];
      console.log("提取到 Task ID:", videoName);
    }

    // 构造webhook请求体（按照飞书webhook格式）
    const webhookData = {
      name: videoName,
      url: requestUrl,
    };

    console.log("准备发送数据:", JSON.stringify(webhookData, null, 2));

    // 发送到webhook（异步，不阻塞请求）
    // 注意：$httpClient是异步的，不会阻塞$done()的调用
    $httpClient.post(
      {
        url: WEBHOOK_URL,
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(webhookData),
      },
      function (error, response, data) {
        if (error) {
          console.log("✗ 发送到webhook失败:", error);
        } else {
          console.log("✓ 发送到webhook成功");
          console.log("响应状态码:", response.status);
          if (data) {
            console.log("响应数据:", data);
          }
        }
      }
    );

    console.log("→ 已触发异步webhook发送");
  }
} else {
  console.log("→ 非 Sora 视频链接，跳过处理");
}

console.log("=== 脚本执行完成 ===");

// 关键：使用 $done({}) 保持请求不变，让请求正常通过
// 这样iOS app可以正常接收和使用视频数据
// $httpClient.post 是异步的，会在后台发送webhook，不影响请求流程
$done({});

复制Raw链接，记下备用 ​

进行ShadowRocket配置 ​

  • 打开ShadowRocket - 配置 - 选择自己的配置文件（默认是default.conf）

  • 找到HTTPS加密，打开

  • 在域名中，输入videos.openai.com，按下左边的+添加

  • 点击开启HTTPS加密，按指示安装证书，信任证书

  • 出现'已下载描述文件'弹窗提示，这时直接返回桌面，打开系统设置 - 通用 - VPN与设备管理

  • 打开系统设置 - 通用 - 关于本机 - 证书信任设置

  • 返回ShadowRocket界面，点击右上角的✓，完成，返回菜单

  • 找到编辑纯文本，打开

  • 滑动到下面，找到[MITM]，在它的上方，填入以下内容，注意替换[上一步记下的Raw链接]为你实际获取到的链接，点击保存
javascript
[Script]
sora-webhook = type=http-request,pattern=^https?://videos\.openai\.com/,requires-body=0,script-path=[上一步记下的Raw链接],debug=1

  • 看到HTTPS解密、脚本、脚本URL都为1，说明配置成功

使用 ​

  • 返回ShadowRocket首页，全局路由改为配置或代理，启用

  • 返回飞书多维表格，打开数据表

  • 然后打开Sora App开始刷视频就行了，看过的视频会自动抓取无水印链接并发送到表格中（没看过的也会，应该是预览机制造成的）。

  • Congratulations!
